<script>
let chart;

document.getElementById("ingreso").oninput = function() {
  document.getElementById("ingresoVal").innerText = this.value;
}

function calcular() {

let ingreso = parseFloat(document.getElementById("ingreso").value);
let vivienda = parseFloat(document.getElementById("vivienda").value);
let cuota = parseFloat(document.getElementById("cuota").value)/100;
let tasa = parseFloat(document.getElementById("tasa").value)/100/12;
let plazo = parseFloat(document.getElementById("plazo").value);
let smmlv = parseFloat(document.getElementById("smmlv").value);

let ingresoSMMLV = ingreso / smmlv;

/* SUBSIDIO */
let subsidio = 0;

if(ingresoSMMLV <= 2){
  subsidio = 30000000;
} else if(ingresoSMMLV <= 4){
  subsidio = 20000000;
}

/* CALCULOS */
let cuotaInicial = vivienda * cuota;
let credito = vivienda - cuotaInicial - subsidio;

/* CUOTA MENSUAL */
let cuotaMensual = credito * (tasa*Math.pow(1+tasa,plazo)) / (Math.pow(1+tasa,plazo)-1);

/* EVALUACION */
let capacidad = ingreso * 0.3;

let estado = "";
let clase = "";

if(cuotaMensual <= capacidad){
  estado = "APROBACIÓN PROBABLE";
  clase = "aprobado";
} else {
  estado = "NO VIABLE";
  clase = "noaprobado";
}

/* RESULTADO */
document.getElementById("resultado").innerHTML = `
<b>Ingreso en SMMLV:</b> ${ingresoSMMLV.toFixed(2)}<br>
<b>Subsidio:</b> $${subsidio.toLocaleString()}<br>
<b>Cuota inicial:</b> $${cuotaInicial.toLocaleString()}<br>
<b>Crédito:</b> $${credito.toLocaleString()}<br>
<b>Cuota mensual:</b> $${cuotaMensual.toFixed(0).toLocaleString()}<br>
<b>Capacidad de pago:</b> $${capacidad.toLocaleString()}<br>
<br>
<span class="${clase}">${estado}</span>
<br><br>
<small>Simulación basada en criterios financieros estándar en Colombia. Sujeta a validación por entidades financieras.</small>
`;

/* GRAFICO */
if(chart) chart.destroy();

chart = new Chart(document.getElementById("grafico"), {
  type: 'line',
  data: {
    labels: ["Inicio","Mitad","Final"],
    datasets: [{
      label: "Deuda",
      data: [credito, credito/2, 0],
      borderWidth: 2,
      fill: false
    }]
  }
});

}
</script>
